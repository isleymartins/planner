const router = require("express").Router();

const cartaController = require("../controllers/cartaController");

router
    .route("/cartas")
    .post((req, res) => cartaController.create(req, res));
router
    .route("/cartas")
    .get((req, res) => cartaController.getAll(req, res));
router
    .route("/cartas/:id")
    .get((req, res) => cartaController.get(req, res));
router
    .route("/cartas/:id")
    .delete((req, res) => cartaController.delete(req, res));
router
    .route("/cartas/:id")
    .put((req, res) => cartaController.update(req, res));
module.exports = router
