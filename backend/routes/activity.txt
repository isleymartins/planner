const router = require("express").Router();

const activityController = require("../controllers/activityController")

router
    .route("/activity")
    .post((req, res) => activityController.create(req, res));
router
    .route("/activity")
    .get((req, res) => activityController.getAll(req, res));
router
    .route("/activity/:id")
    .get((req, res) => activityController.get(req, res));
router
    .route("/activity/:id")
    .delete((req, res) => activityController.delete(req, res));
router
    .route("/activity/:id")
    .put((req, res) => activityController.update(req, res));
module.exports = router
